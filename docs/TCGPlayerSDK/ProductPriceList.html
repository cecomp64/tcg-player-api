def initialize(hash = {})
  super

  if(self.success && self.results)
    @prices = self.results.map{|r| TCGPlayerSDK::ProductPrice.new(r)}.group_by{|r| r.productId}
  else
    @prices = {}
  end
end

def valid_prices
  valid_prices = {}
  @prices.each do |id, prl|
    valid_prices[id] ||= []
    valid_prices[id] = prl.select{|pr| pr.has_valid_prices?}
  end

  return valid_prices
end
